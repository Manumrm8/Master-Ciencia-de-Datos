Accuracy con PCA y LDA

In[]:
pca=PCA()
pca.fit_transform(X_tr)

cumsum_var_ratio = np.cumsum(pca.explained_variance_ratio_)

# Graficar la suma acumulativa de los valores singulares
plt.plot(cumsum_var_ratio, linestyle='-', color='b')
plt.xlabel('Nmero de Componentes Principales')
plt.ylabel('Suma Acumulativa de la Varianza Explicada')
plt.title('Anlisis de Componentes Principales (PCA) - Cumsum')
plt.grid(True)
plt.show()
In[]:
